result = []
void backtrack(路径, 选择列表){
    if(满足结束条件){
        result.add(路径)
        return
    }
    for(选择:选择列表){               //搜索树的宽度
        做选择
        backtrack(路径, 选择列表)     //搜索树的深度，路径与选择列表分别给出了搜索目标与搜索范围（0还是index）
        撤销选择
    }
}
